79 function draw(){
80  var canvas = document.getElementById('target');
81 
82  if(canvas.getContext){ //It exists. It should always exist, mind you.
83  var ctx = canvas.getContext('2d'); //canvas context, the thing we draw into
84 
85  //get the current values of p, q, and qhat
86  var p = document.getElementById('p').value;
87  var q = document.getElementById('q').value;
88  var qhat = q/gcd(p, q);
89  document.getElementById('tile').value = qhat;
90 
91  //check for smaller tile size from quadratic residues
92  if(document.getElementById('ornamentation').checked)
93  for(var r=2; r<=qhat/2; r++)
94  if((r*r)%q == 0){ //if r is a quadratic residue mod q
95  var p2 = 2*p*p;
96  if(p2%r == 0){ //if r divides 2p^2
97  qhat = r; //ensures that we exit the for loop
98  document.getElementById('tile').value = r; //new tile size
99  } //if
100  } //if
101 
102  //set the canvas resolution to 800x600, unless we're displaying a single tile
103  if(document.getElementById('showtilecheckbox').checked && qhat <= 600) //single tile
104  ctx.canvas.width = ctx.canvas.height = qhat;
105  else{ //default
106  ctx.canvas.width = 800;
107  ctx.canvas.height = 600;
108  } //else
109 
110  //get the pixel data array from the canvas context
111  var myImageData = ctx.getImageData(0, 0, canvas.width, canvas.height);
112  var data = myImageData.data;
113 
114  //figure out how many color bands to use
115  var bandseditbox = document.getElementById('bands');
116  bandseditbox.value = Math.min(Math.max(1, bandseditbox.value), 4); //clip to bounds
117  var bands = bandseditbox.value; //shorthand
118 
119  //shorthands for width, height
120  var w = canvas.width;
121  var h = canvas.height;
122 
123  //now we can actually draw the texture
124 
125  //tile size is bigger than the canvas, so compute the whole thing
126  if(qhat > w || qhat > h){
127  var px = 0; //pixel pointer
128 
129  for(var i=0; i<h; i++)
130  for(var j=0; j<w; j++)
131  px = setPixel(data, px, colorize(pi(j, q - i), bands));
132 
133  ctx.putImageData(myImageData, 0, 0); //draw the texture
134  } //if
135 
136  else{ //it's quicker to compute one tile and copy it across the whole canvas
137  var tile = ctx.createImageData(qhat, qhat); //create image for tile
138  data = tile.data; //drawing into tile now
139  var px = 0; //pixel pointer
140 
141  for(var i=0; i<qhat; i++)
142  for(var j=0; j<qhat; j++)
143  px = setPixel(data, px, colorize(pi(j, qhat - i), bands));
144 
145  //copy tile across the texture
146  var x=0, y=0; //coordinates in tiles
147 
148  while(y < h){ //for each row
149  while(x < w){ //for each column
150  ctx.putImageData(tile, x, y); //draw the tile
151  x += qhat; //next column
152  } //while
153  x = 0; y += qhat; //next row, first column
154  } //while
155  } //else
156  } //if
157 
158  DisplayInstructions(); //display instructions at the bottom of the window
159 } //draw
